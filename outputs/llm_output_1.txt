colvar {
  name t_goldschmidt
  width 0.01
  lowerBoundary 0.5
  upperBoundary 1.2

  atomGroup {
    name A_cations
    atomNumbers 1 2 3 4 5
  }

  atomGroup {
    name B_cations
    atomNumbers 6 7 8 9 10
  }

  atomGroup {
    name X_anions
    atomNumbers 11 12 13 14 15 16 17 18
  }

  distance {
    name r_A_X
    group1 { atomGroup A_cations }
    group2 { atomGroup X_anions }
  }

  distance {
    name r_B_X
    group1 { atomGroup B_cations }
    group2 { atomGroup X_anions }
  }

  # Goldschmidt tolerance factor: t = (r_A + r_X) / sqrt(2)*(r_B + r_X)
  # Here approximated by ratio of distances as proxy
  # Define a custom colvar as ratio of distances (A-X)/(sqrt(2)*B-X)
  # Colvars module does not support arithmetic directly, so define two distances as CVs
}

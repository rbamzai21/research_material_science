```tcl
# colvars.conf

# Collective Variable 1: Average B-X bond length
atomGroup {
  name b_atoms
  atomNumbers 101 102 103 104 105 106
}
atomGroup {
  name x_atoms
  atomNumbers 201 202 203 204 205 206 207 208
}
colvar {
  name avg_bx_bond_length
  width 0.01
  distance {
    group1 { atomGroup b_atoms }
    group2 { atomGroup x_atoms }
    # Calculate average over all B-X pairs within a cutoff
    cutoff 3.0
  }
  average {
    # Average over all pairs found
    overPairs true
  }
}

# Collective Variable 2: BX6 octahedral distortion parameter
# Defined as the standard deviation of B-X bond lengths
colvar {
  name bx_octahedral_distortion
  width 0.001
  distance {
    group1 { atomGroup b_atoms }
    group2 { atomGroup x_atoms }
    cutoff 3.0
  }
  stddev {
    overPairs true
  }
}

# Collective Variable 3: A-site cation coordination number with X anions
atomGroup {
  name a_atoms
  atomNumbers 1 2 3 4 5 6
}
coordinationNumber {
  name a_x_coordination
  group1 { atomGroup a_atoms }
  group2 { atomGroup x_atoms }
  cutoff 3.5
}
```
